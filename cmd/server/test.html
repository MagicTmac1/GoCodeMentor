GMP模型说明

G
Goroutine (G): 轻量级线程，由Go运行时管理，创建成本低，栈大小可动态调整
M
Machine (M): 操作系统线程，负责执行Goroutine，数量可动态调整
P
Processor (P): 虚拟处理器，连接G和M，数量通常等于CPU核心数

GMP模型详解

调度原理

1. 调度器设计目标：充分利用多核性能，同时简化并发编程模型

2. 工作窃取机制：当一个P的本地队列为空时，会从其他P的队列或全局队列窃取Goroutine

3. 抢占式调度：运行超过10ms的Goroutine会被标记为可抢占，在系统调用返回时触发调度

4. 阻塞恢复：当M被阻塞时，P会寻找其他空闲的M或创建新的M来执行队列中的Goroutine

5. 调度策略：优先从本地队列获取Goroutine，然后是全局队列，最后尝试工作窃取

优势特点

轻量级
每个Goroutine仅占2KB内存
高并发
单进程可支持上百万Goroutine
低延迟
调度延迟通常在微秒级别
自动伸缩
根据负载动态调整线程数

Go并发编程示例

package main

import (
    "fmt"
    "time"
)

func worker(id int, jobs <-chan int, results chan<- int) {
    for j := range jobs {
        fmt.Println("worker", id, "started job", j)
        time.Sleep(time.Second)
        fmt.Println("worker", id, "finished job", j)
        results <- j * 2
    }
}

func main() {
    const numJobs = 5
    
    // 创建任务和结果通道
    jobs := make(chan int, numJobs)
    results := make(chan int, numJobs)
    
    // 启动3个worker协程
    for w := 1; w <= 3; w++ {
        go worker(w, jobs, results)
    }
    
    // 发送任务
    for j := 1; j <= numJobs; j++ {
        jobs <- j
    }
    close(jobs)
    
    // 收集结果
    for a := 1; a <= numJobs; a++ {
        <-results
    }
}
package main

import (
    "fmt"
    "time"
)

func main() {
    // 创建无缓冲通道
    ch := make(chan string)
    
    // 启动协程发送数据
    go func() {
        for i := 0; i < 5; i++ {
            ch <- fmt.Sprintf("消息 %d", i)
            time.Sleep(500 * time.Millisecond)
        }
        close(ch) // 关闭通道
    }()
    
    // 从通道接收数据
    for msg := range ch {
        fmt.Println("接收到:", msg)
    }
    
    // 使用select处理多个通道
    ch1 := make(chan string)
    ch2 := make(chan string)
    
    go func() {
        time.Sleep(1 * time.Second)
        ch1 <- "来自通道1"
    }()
    
    go func() {
        time.Sleep(2 * time.Second)
        ch2 <- "来自通道2"
    }()
    
    for i := 0; i < 2; i++ {
        select {
        case msg1 := <-ch1:
            fmt.Println("收到:", msg1)
        case msg2 := <-ch2:
            fmt.Println("收到:", msg2)
        case <-time.After(1500 * time.Millisecond):
            fmt.Println("超时")
        }
    }
}
package main

import (
    "fmt"
    "sync"
    "time"
)

// 共享资源
type Counter struct {
    mu    sync.Mutex
    value int
}

// 安全地增加计数
func (c *Counter) Increment() {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.value++
}

// 安全地获取计数值
func (c *Counter) Value() int {
    c.mu.Lock()
    defer c.mu.Unlock()
    return c.value
}

func main() {
    counter := &Counter{}
    var wg sync.WaitGroup
    
    // 启动10个协程，每个协程增加计数1000次
    for i := 0; i < 10; i++ {
        wg.Add(1)
        go func(id int) {
            defer wg.Done()
            for j := 0; j < 1000; j++ {
                counter.Increment()
                time.Sleep(time.Microsecond) // 模拟工作
            }
            fmt.Printf("协程 %d 完成\n", id)
        }(i)
    }
    
    // 等待所有协程完成
    wg.Wait()
    
    fmt.Printf("最终计数: %d\n", counter.Value())
    
    // 使用RWMutex进行读写锁演示
    var rwMutex sync.RWMutex
    sharedData := 0
    
    // 多个读协程
    for i := 0; i < 5; i++ {
        wg.Add(1)
        go func(id int) {
            defer wg.Done()
            for j := 0; j < 3; j++ {
                rwMutex.RLock() // 读锁
                fmt.Printf("读协程 %d 读取数据: %d\n", id, sharedData)
                rwMutex.RUnlock()
                time.Sleep(100 * time.Millisecond)
            }
        }(i)
    }
    
    // 写协程
    wg.Add(1)
    go func() {
        defer wg.Done()
        for i := 0; i < 3; i++ {
            rwMutex.Lock() // 写锁
            sharedData++
            fmt.Printf("写协程更新数据: %d\n", sharedData)
            rwMutex.Unlock()
            time.Sleep(300 * time.Millisecond)
        }
    }()
    
    wg.Wait()
}
package main

import (
    "fmt"
    "sync"
    "time"
)

// 工作池
type WorkerPool struct {
    workers    int
    jobQueue   chan Job
    resultQueue chan Result
    wg         sync.WaitGroup
}

type Job struct {
    ID   int
    Data int
}

type Result struct {
    Job    Job
    Output int
}

// 创建新的工作池
func NewWorkerPool(workers int) *WorkerPool {
    return &WorkerPool{
        workers:    workers,
        jobQueue:   make(chan Job, 100),
        resultQueue: make(chan Result, 100),
    }
}

// 启动工作池
func (wp *WorkerPool) Start() {
    for i := 0; i < wp.workers; i++ {
        wp.wg.Add(1)
        go wp.worker(i)
    }
}

// 工作函数
func (wp *WorkerPool) worker(id int) {
    defer wp.wg.Done()
    for job := range wp.jobQueue {
        fmt.Printf("工作者 %d 处理任务 %d\n", id, job.ID)
        // 模拟工作
        time.Sleep(100 * time.Millisecond)
        result := Result{
            Job:    job,
            Output: job.Data * job.Data,
        }
        wp.resultQueue <- result
    }
}

// 添加任务
func (wp *WorkerPool) AddJob(job Job) {
    wp.jobQueue <- job
}

// 关闭工作池
func (wp *WorkerPool) Stop() {
    close(wp.jobQueue)
    wp.wg.Wait()
    close(wp.resultQueue)
}

func main() {
    // 创建8个工作者的工作池
    pool := NewWorkerPool(8)
    pool.Start()
    
    // 添加100个任务
    for i := 0; i < 100; i++ {
        pool.AddJob(Job{
            ID:   i,
            Data: i,
        })
    }
    
    // 关闭工作池并等待完成
    go func() {
        pool.Stop()
    }()
    
    // 收集结果
    results := make(map[int]int)
    for result := range pool.resultQueue {
        results[result.Job.ID] = result.Output
        fmt.Printf("任务 %d 完成，结果: %d\n", result.Job.ID, result.Output)
        
        // 当所有结果都收集到了，退出
        if len(results) == 100 {
            break
        }
    }
    
    fmt.Println("所有任务完成!")
}
